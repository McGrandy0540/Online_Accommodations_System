# Run this command in Command Prompt/Terminal:
mysqldump -u root -p online_accommodations_system > online_accommodations_system_complete_backup.sql

# METHOD 2: Export only database structure (no data)
mysqldump -u root -p --no-data online_accommodations_system > online_accommodations_system_structure_only.sql

# METHOD 3: Export only data (no structure)
mysqldump -u root -p --no-create-info online_accommodations_system > online_accommodations_system_data_only.sql

# METHOD 4: Export specific tables
mysqldump -u root -p online_accommodations_system users properties rooms bookings > specific_tables_backup.sql

# METHOD 5: Export with additional options (recommended)
mysqldump -u root -p --single-transaction --routines --triggers --events online_accommodations_system > online_accommodations_system_full_backup.sql

# METHOD 6: Export compressed backup
mysqldump -u root -p online_accommodations_system | gzip > online_accommodations_system_backup.sql.gz

# =====================================================
# AUTOMATED BACKUP SCRIPT (Windows Batch)
# =====================================================
# Create a .bat file with this content:
@echo off
set MYSQL_PATH="C:\Program Files\MySQL\MySQL Server 8.0\bin"
set BACKUP_PATH="C:\backup"
set DATE=%date:~-4,4%%date:~-10,2%%date:~-7,2%
set TIME=%time:~0,2%%time:~3,2%%time:~6,2%
set FILENAME=online_accommodations_system_%DATE%_%TIME%.sql

%MYSQL_PATH%\mysqldump -u root -p online_accommodations_system > %BACKUP_PATH%\%FILENAME%
echo Backup completed: %FILENAME%
pause
